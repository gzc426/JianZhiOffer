package sy181129;

/**
 * @author suyuan
 *	一只青蛙一次可以跳上1级台阶，也可以跳上2级台阶。
 *	求该青蛙跳上一个n级台阶总共有多少种跳法。
 *
 *	f(1)=1 f(2)=2
 *	f(3) 1+1+1 1+2 2+1 =3
 *	f(4) 1+1+1+1 1+1+2 1+2+1 2+1+1 2+2 =5
 */
public class jz_跳台阶
{

	public static void main(String[] args)
	{
		System.out.println(tiao1(4));
		System.out.println(tiao1(5));
		System.out.println(tiao2(4));
		System.out.println(tiao2(5));

	}
	
	public static int tiao1(int n)
	{
		if(n==1)
			return 1;
		if(n==2)
			return 2;
		int tmp1=1;
		int tmp2=2;
		int sum=tmp1+tmp2;
		for(int i=3;i<=n;i++)
		{
			sum=tmp1+tmp2;
			tmp1=tmp2;
			tmp2=sum;
		}
		return sum;
	}
	
	public static int tiao2(int n)
	{
		if(n<0)
			return -1;
		if(n==1 || n==2)
			return 1;
		int[][] base={ { 1, 1 },{ 1, 0 } };
		int[][] res=matrixPower(base, n-2);
		return 2*res[0][0]+res[1][0];
	}
	
	public static int[][] matrixPower(int [][] m,int p)
	{
		int[][] res=new int[m.length][m[0].length];
		 //把res设置为单位矩阵，类似于整数中的1；
		for(int i=0;i<res.length;i++)
		{
			res[i][i]=1;
		}
		int [][] tmp=m;
		for(;p!=0;p>>=1)
		{
			if((p&1)!=0)
				res=muliMatrix(res, tmp);//奇数
			//else {
			tmp=muliMatrix(tmp, tmp);
			//}
		
		}
	
	
		return res;
	}
	
	public static int[][] muliMatrix(int [][] m1,int [][] m2)
	{
		int[][] res=new int[m1.length][m2[0].length];
		for(int i=0;i<m2[0].length;i++)
		{
			for(int j=0;j<m1.length;j++)
			{
				for(int k=0;k<m2.length;k++)
				{
					res[i][j]+=m1[i][k]*m2[k][j];
				}
			}
		}
		
		return res;
	}
	

}
